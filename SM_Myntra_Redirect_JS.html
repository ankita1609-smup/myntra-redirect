<script>

    // Function to get URL parameters

    function getQueryParam(param) {

        let params = new URLSearchParams(window.location.search);

        return params.get(param);

    }


    // Get the Myntra product URL from query parameter

    let myntraProductURL = getQueryParam("url");


    // If no product URL is provided, redirect to Myntra homepage

    if (!myntraProductURL) {

        myntraProductURL = "https://www.myntra.com";

    }


    // Attempt to open in Myntra app

    window.location.href = "myntra://openapp?url=" + myntraProductURL;


    // Fallback to browser after 2 seconds if app doesn't open

    setTimeout(function () {

        window.location.href = myntraProductURL;

    }, 2000);

</script>
